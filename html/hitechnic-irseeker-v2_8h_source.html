378 bool HTIRS2readEnhanced(tSensors link, int &dir, int &strength)
379 {
380  ubyte iMax = 0;
381  long dcSigSum = 0;
382 
383  // Read DC mode
384  memset(HTIRS2_I2CRequest, 0, sizeof(tByteArray));
385 
386  HTIRS2_I2CRequest[0] = 2; // Message size
387  HTIRS2_I2CRequest[1] = HTIRS2_I2C_ADDR; // I2C Address
388  HTIRS2_I2CRequest[2] = HTIRS2_OFFSET + HTIRS2_DC_SSTR1; // Sensor 0 signal strength
389 
390  if (!writeI2C(link, HTIRS2_I2CRequest, HTIRS2_I2CReply, 6))
391  return false;
392 
393  // Find the max DC sig strength
394  for (int i = 1; i < 5; i++)
395  {
396  if (HTIRS2_I2CReply[i] > HTIRS2_I2CReply[iMax])
397  {
398  iMax = i;
399  }
400  }
401 
402  // Calc base DC direction value
403  dir = iMax * 2 + 1;
404  // Set base dcStrength based on max signal and average
405  dcSigSum = HTIRS2_I2CReply[iMax] + HTIRS2_I2CReply[5];
406 
407  // Check signal strength of neighbouring sensor elements
408  if ((iMax > 0) && (HTIRS2_I2CReply[iMax - 1] > (ubyte)(HTIRS2_I2CReply[iMax] / 2)))
409  {
410  dir--;
411  dcSigSum += HTIRS2_I2CReply[iMax - 1];
412  }
413 
414  if ((iMax < 4) && (HTIRS2_I2CReply[iMax + 1] > (ubyte)(HTIRS2_I2CReply[iMax] / 2)))
415  {
416  dir++;
417  dcSigSum += HTIRS2_I2CReply[iMax + 1];
418  }
419 
420  // Make DC strength compatible with AC strength.
421  // use: sqrt(dcSigSum*500)
422  strength = sqrt(dcSigSum * 500);
423 
424  // Decide if using AC strength
425  if (strength <= 200)
426  {
427  writeDebugStreamLine("switching to AC");
428  // Use AC Dir
429  HTIRS2_I2CRequest[2] = HTIRS2_OFFSET + HTIRS2_AC_DIR; // Recycle rest of cmdBuf
430 
431  if (!writeI2C(link, HTIRS2_I2CRequest, HTIRS2_I2CReply, 6))
432  return false;
433 
434  dir = HTIRS2_I2CReply[0];
435 
436  // Sum the sensor elements to get strength
437  if (dir > 0)
438  {
439  strength = HTIRS2_I2CReply[1] + HTIRS2_I2CReply[2] +
440  HTIRS2_I2CReply[3] + HTIRS2_I2CReply[4] +
441  HTIRS2_I2CReply[5];
442  }
443  }
444  return true;
445 }
